package com.leetcode;

/**
 * Algorithm
 * Loop
 * 1. carry => ~a & b
 * 2. a => a ^ b
 * 3. b => carry << 1
 */
public class Program {
    public static int getSubtract(int a, int b) {
        if(a == 0) return b;
        if(b == 0) return a;

        while (b != 0) {
            int carry = ~a & b;
            a = a ^ b;
            b = carry << 1;
        }
        return a;
    }

    public static void main(String[] args) {
        System.out.println(getSubtract(2,4));
    }
}
