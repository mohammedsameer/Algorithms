import java.io.*;
import java.util.*;

/*
 * To execute Java, please define "static void main" on a class
 * named Solution.
 *
 * If you need more classes, simply define them inline.
 */

class Solution {
  
  public static boolean search(int[] arr,int target)
  {
    if(arr.length==0)
      return false;
    
    for(int i=0;i<arr.length;i++)
    {
      if(target==arr[i])
        return true;
      
     //rotation encountered & target still not matching/smaller
     if(i<arr.length-1 && arr[i]>arr[i+1])
      {
        if(target>arr[i])
          return false;
      }
      
    }
    
    return false;
    
  }
  
  public static void main(String[] args) {
   int[] arr = {3,4,5,5,1,1,2};
   System.out.println("Search in Rotated Sorted Array: "+ search(arr, 3));
  }
  
}
