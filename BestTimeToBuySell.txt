import java.io.*;
import java.util.*;

/*
 * To execute Java, please define "static void main" on a class
 * named Solution.
 *
 * If you need more classes, simply define them inline.
 */

class Solution {
  
  public static int bestTimeToBuy(int[] stockPrices) {
    
    if(stockPrices.length==0)
      return -1;
    
    int profit = 0;
    int min = Integer.MAX_VALUE;
    
    for(int i=0;i<stockPrices.length;i++)
    {
      min = Math.min(stockPrices[i], min);
      profit = Math.max(profit, stockPrices[i]-min);
    }
    
   return profit;
  }
  
  public static void main(String[] args) {
    int [] stockPrices = {90,100,20,10};    
   System.out.println("Maximum Profit: "+ bestTimeToBuy(stockPrices));
  }
  
}
