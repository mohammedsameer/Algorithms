package com.leetcode;

import java.util.*;

/**
 * Algorithm
 * 1. Store current array in set
 * 2. Loop through array and see if target - diff is present in set
 */
public class Program {

    private static int[] twoSum(int[] inputArr, int target) {
        Set<Integer> inputSet = new HashSet<>();
        int[] result = {-1, -1};

        for(int num : inputArr) {
            inputSet.add(num);
        }

        for(int num : inputArr) {
            int diff = target - num;
            if(inputSet.contains(diff)) {
                result[0] = num;
                result[1] = diff;
                return result;
            }
        }

        return result;
    }

    public static void main(String[] args) {
        int[] inputArr = {0, 5, 3, 8, 9};
        int target = 5;
        int[] result = twoSum(inputArr, target);
        for(int entity: result) {
            System.out.print(entity + " ");
        }
    }
}
