import java.io.*;
import java.util.*;

/*
 * To execute Java, please define "static void main" on a class
 * named Solution.
 *
 * If you need more classes, simply define them inline.
 */

class Solution {
  public static ArrayList<ArrayList<Integer>> pascalTriangle(int num)
  {
    if(num<=0)
      return null;
    
    ArrayList<ArrayList<Integer>> result = new ArrayList<ArrayList<Integer>>();
    ArrayList<Integer> curr = new ArrayList<>();
    ArrayList<Integer> prev = new ArrayList<>();
    curr.add(1);
    result.add(curr);
    
    for(int line=2;line<=num;line++)
    {
      prev = result.get(line-2);
      curr = new ArrayList<>();
      
      for(int i=0;i<line;i++)
      {
        int val = 0;
        if(i==0||i==line-1)
        {
          val = 1;
        }
        else
        {
          val = prev.get(i-1)+prev.get(i);
        }
        curr.add(val);
      }
      
      result.add(curr);
    }
    
    return result;
    
  }
  
  public static void main(String[] args) {
  System.out.println("Pascal Triangle: "+ pascalTriangle(3));
  }
}
