import java.io.*;
import java.util.*;

/*
 * To execute Java, please define "static void main" on a class
 * named Solution.
 *
 * If you need more classes, simply define them inline.
 */

class Solution {
  
  public static ArrayList<ArrayList<Integer>> permute(int[] num)
  {
    ArrayList<ArrayList<Integer>> result = new ArrayList<ArrayList<Integer>>();
    ArrayList<ArrayList<Integer>> currResult = new ArrayList<ArrayList<Integer>>();
    
    if(num.length==0)
      return result;
    
    //Initialize with first element
    result.add(new ArrayList<Integer>());
    
    
    for(int i=0;i<num.length;i++)
    {
      //Insert number in every element at every location in every element of result
      for(ArrayList<Integer> prev: result)
      {
        //Indexed from 0 to len, as we need to add to the end
        for(int j=0;j<=prev.size();j++)
        {
          ArrayList<Integer> curr = new ArrayList<>(prev);
          curr.add(j,num[i]);
          currResult.add(curr);
        }
      }
      
      result = currResult;
      currResult = new ArrayList<ArrayList<Integer>>();
    }
    
    return result;
    
  }
  
  public static void main(String[] args) {
   int[] num = {1,2,3};
   System.out.println("Permutation:"+ permute(num));
  }
}
