import java.io.*;
import java.util.*;

/*
 * To execute Java, please define "static void main" on a class
 * named Solution.
 *
 * If you need more classes, simply define them inline.
 */
//Algo
//1. For every iteration loop through all elements from previous //and insert the new element
class Solution {
  public static ArrayList<ArrayList<String>> subsets(String[] sarr){
    
    ArrayList<ArrayList<String>> currResult = new ArrayList<ArrayList<String>>();
    ArrayList<ArrayList<String>> prevResult = new ArrayList<ArrayList<String>>();
    prevResult.add(new ArrayList<String>());
    
    if(sarr.length==0)
      return prevResult;
    
    for(int i=0;i<sarr.length;i++)
    {
      String s = sarr[i];
      //Initialize currResult with prevResult & loop through to add new element across all elements
      currResult = new ArrayList<ArrayList<String>>(prevResult);
      for(ArrayList<String> prevEntry: prevResult)
      {
        ArrayList<String> currEntry = new ArrayList<String>(prevEntry);
        currEntry.add(s);
        currResult.add(currEntry);
      }
      
      //At the end of iteration reset prevResult with currResult
      prevResult = new ArrayList<ArrayList<String>>(currResult);
    }
    
    return currResult;
  }
  
  public static void main(String[] args) {
    String[] sarr= {"a","b","c"};
    System.out.println("Subsets"+ subsets(sarr));
  }
}
