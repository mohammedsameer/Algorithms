import java.io.*;
import java.util.*;

class Solution {
  
  //isPalindrome:Checks if input is palindrome or not
	public static boolean isPalindrome(String s)
  {
    if(s==null || s.length()==0)
     return false;
    
    //Remove unwanted characters
    s = s.replaceAll("[^a-zA-Z0-9]","").toLowerCase();
    
    int mid = s.length()/2;
    
    for(int i=0; i<mid; i++)
    {
      if(s.charAt(i)!=s.charAt(s.length()-i-1))
      {
         return false;
      }
    }
    
    return true;
  }
  
  public static void main(String[] args) {
    System.out.println("IsPalindrome- mom: "+ isPalindrome("mom"));
    System.out.println("IsPalindrome- mommy: "+ isPalindrome("mommy"));
  }
}
