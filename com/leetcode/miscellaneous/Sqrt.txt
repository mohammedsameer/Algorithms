package com.leetcode;

/**
 * Algorithm
 * The sqrt should be between 1...n/2 for 1....n
 */
public class Program {

    public static int mySqrt(int n) {
        if(n==1)
            return 1;

        int start = 1;
        int end = n;
        int mid = n/2;

        while(start <= end) {
            //If square of mid = n, then mid is sqrt of n
            int square = mid * mid;
            if(square == n)
                return mid;

            if(square < n)
                start = mid + 1;
            else
                end = mid - 1;

            mid = (start + end)/2;
        }

        return mid;
    }

    public static void main(String[] args) {
        System.out.println(mySqrt(119));
    }
}
