public class Solution {
    public ArrayList<Integer> inorderTraversal(TreeNode root){
        
        ArrayList<Integer> result = new ArrayList<>();
        Stack<TreeNode> stack = new Stack<>();
        
        if(root==null)
         return result;
        
        TreeNode p = root;
        
        while(!stack.empty() || p!=null)
        {
            if(p!=null)
            {
              stack.push(p);
              p=p.left;
            }
            else
            {
                TreeNode node = stack.pop();
                result.add(node.val);
                p=node.right;
            }
        }
        
        return result;
    }
}
