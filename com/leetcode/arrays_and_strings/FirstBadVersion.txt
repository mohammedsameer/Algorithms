//Binary Search. The result is in the start after completion
public int firstBadVersion(int n) {
    int start = 1;
    int end = n;

    while(start<=end) {
        int mid = start + (end-start) / 2;
        if(isBadVersion(mid)) {
            //Search for bad version
            end = mid-1;
        } else {
            //No bad versions yet
            start = mid+1;
        }
    }

    return start;
}
