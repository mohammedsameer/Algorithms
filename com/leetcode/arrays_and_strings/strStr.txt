import java.io.*;
import java.util.*;
import java.lang.*;

/*
 * To execute Java, please define "static void main" on a class
 * named Solution.
 *
 * If you need more classes, simply define them inline.
 */

class Solution {
  
  public static boolean strStr(String needle, String hayStack)
  {
    
    if(hayStack==null || needle==null)
      return false;
    
    if(hayStack.length()<needle.length())
      return false;
    
    int hayStackLen = hayStack.length();
    int needleLen = needle.length();
    int diff = hayStackLen - needleLen;
    for(int i = 0; i< diff; i++)
    {
      String sub = hayStack.substring(i, i+needleLen);
      if(sub.equals(needle))
      {  
        return true;
      }
    }
    
    return false;
    
  }
  
  public static void main(String[] args) {
   String needle = "sameer";
   String hayStack="jjhjnLXNsamera";
   System.out.println("isSubStr:"+  strStr(needle, hayStack));
  }
}
