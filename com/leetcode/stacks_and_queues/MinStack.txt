package com.leetcode;

import java.util.*;

class Program {

    public Stack<Integer> stack = new Stack<>();
    public Stack<Integer> minStack = new Stack<>();

    public void push(int e){
        stack.add(e);
        if((!minStack.isEmpty() && minStack.peek() > e) || minStack.isEmpty())
        {
            minStack.add(e);
        }

    }

    public int pop() {
        int value = -1;
        if(!stack.isEmpty()) {
            value = stack.pop();
            if(!minStack.isEmpty() && minStack.peek() == value) {
                minStack.pop();
            }
        }

        return value;
    }

    public int getMin() {
        int value = -1;
        if(!minStack.isEmpty()) {
            value = minStack.peek();
        }
        return value;
    }

    public int top(){
        if(!stack.isEmpty()) {
            return stack.peek();
        }

        return -1;
    }

    public static void main(String[] args) {

    }
}
